实现一个高级表单组件（ProForm），要求能支持如下两种方式来渲染表单。

1. json定义：

```tsx

  import { Upload } from 'antd'
  import { CURRENCY_TYPE_ENUM } from '@/apis/enum'
  import AddressWidget from '@/components/address-widget'

  const TIER2 = [CURRENCY_TYPE_ENUM.NGN, CURRENCY_TYPE_ENUM.TZS, CURRENCY_TYPE_ENUM.KES]

  const columns: ProFormColumnsType[] = [
    {
      title: 'accountDetail.form.title',
      valueType: 'group',
      columns: [],
      formItemProps: { className: 'text-16px font-bold mb-4' }, // 纯 UI
    },
    {
      title: 'accountDetail.form.accountType',
      dataIndex: 'bankAccountType',
      valueType: 'select',
      fieldProps: {
        placeholder: 'accountDetail.form.accountTypePlaceholder',
      },
      formItemProps: {
        rules: [{ required: true, message: 'accountDetail.form.accountTypeRequired' }],
      },
      valueEnum: 'accountTypeOptions', // 由外部 dataSource 解析
    },
    {
      title: 'accountDetail.form.currency',
      dataIndex: 'currencyType',
      valueType: 'select',
      fieldProps: {
        placeholder: 'accountDetail.form.currencyPlaceholder',
        disabled: true,
      },
      valueEnum: 'currencyOptions',
      formItemProps: { rules: [{ required: true }] },
    },
    {
      title: 'accountDetail.form.accountName',
      dataIndex: 'accountName',
      valueType: 'text',
      fieldProps: { placeholder: 'accountDetail.form.accountNamePlaceholder' },
      formItemProps: { rules: [{ required: true }] },

    {
      title: 'accountDetail.form.accountNumber',
      dataIndex: 'cardNo',

    },

      valueType: 'dependency',

      columns: ({ currencyType }) =>
        TIER2.includes(currencyType)
          ? []
          : [
              {
                title: 'accountDetail.form.countryRegion',
                dataIndex: 'accAddrCountry',
                valueType: 'select',
                valueEnum: { type: 'api', key: 'countryList', extraMap: { label: 'name', value: 'code' } },
                fieldProps: { placeholder: 'accountDetail.form.countryRegionPlaceholder' },
                formItemProps: { rules: [{ required: true }] },
              },
              {
                title: 'accountDetail.form.bankAddress',
                dataIndex: 'bankAddress',
                valueType: 'text',
                fieldProps: { placeholder: 'accountDetail.form.bankAddressPlaceholder' },
                formItemProps: { rules: [{ required: true }] },
              },
              {
                title: 'accountDetail.form.swift',
                dataIndex: 'bankSwift',
                valueType: 'text',
                fieldProps: { placeholder: 'accountDetail.form.swiftPlaceholder' },

              { title: 'accountDetail.form.iban', dataIndex: 'iban', valueType: 'text' },
              { title: 'accountDetail.form.bsbNumber', dataIndex: 'bsbNumber', valueType: 'text' },
              { title: 'accountDetail.form.routingNumber', dataIndex: 'routingNumber', valueType: 'text' },

                title: 'accountDetail.form.recipientAddress',
                dataIndex: 'recipientAddress',
                valueType: 'form',
```

2. jsx定义：

```tsx
  import { ProForm, ProFormDependency, ProFormDigit, ProFormText, ProFormSelect } from '@/components/pro-form'
  import { Form, Upload, Button } from 'antd'
  import { CURRENCY_TYPE_ENUM } from '@/apis/enum'
  import AddressWidget from '@/components/address-widget'

  const TIER2 = [CURRENCY_TYPE_ENUM.NGN, CURRENCY_TYPE_ENUM.TZS, CURRENCY_TYPE_ENUM.KES]

  export default function BankAccountProForm({
    accountTypeOptions,
    countryList,
    bankList,
    defaultValues,
    onSubmit,
  }) {
    const [form] = Form.useForm()

    const countryOptions = countryList?.data?.map(item => ({ label: item.name, value: item.code })) ?? []
    const bankOptions = bankList?.data?.map(item => ({ label: item.bankName, value: item.bankCode })) ?? []

    return (
      <ProForm
        form={form}
        layout="vertical"
        initialValues={defaultValues}
        submitter={{
          render: (_, dom) => (
            <div className="flex justify-end">{dom}</div>
          ),
        }}
        onFinish={onSubmit}
      >
        <div className="text-16px font-bold mb-4">{t('accountDetail.form.title')}</div>

        <ProFormSelect
          name="bankAccountType"
          label={t('accountDetail.form.accountType')}
          placeholder={t('accountDetail.form.accountTypePlaceholder')}
          options={accountTypeOptions}
          rules={[{ required: true, message: t('accountDetail.form.accountTypeRequired') }]}
        />

        <ProFormSelect
          name="currencyType"
          label={t('accountDetail.form.currency')}
          placeholder={t('accountDetail.form.currencyPlaceholder')}
          options={CURRENCY_OPTIONS}
          disabled
          rules={[{ required: true }]}
        />

        <ProFormText
          name="accountName"
          label={t('accountDetail.form.accountName')}
          placeholder={t('accountDetail.form.accountNamePlaceholder')}
          rules={[{ required: true }]}
        />

        <ProFormText
          name="cardNo"
          label={t('accountDetail.form.accountNumber')}
          placeholder={t('accountDetail.form.accountNumberPlaceholder')}
          rules={[{ required: true }]}
        />

        {/* 银行名称切换 input/select */}
        <ProFormDependency name={['currencyType']}>
          {({ currencyType }) => {
            const isTier2 = TIER2.includes(currencyType)
            return isTier2
              ? (
                  <ProFormSelect
                    name="bankName"
                    label={t('accountDetail.form.bankName')}
                    placeholder={t('accountDetail.form.bankNamePlaceholder')}
                    showSearch
                    options={bankOptions}
                    fieldProps={{ optionFilterProp: 'label' }}
                    rules={[{ required: true }]}
                  />
                )
              : (
                  <ProFormText
                    name="bankName"
                    label={t('accountDetail.form.bankName')}
                    placeholder={t('accountDetail.form.bankNamePlaceholder')}
                    rules={[{ required: true }]}
                  />
                )
          }}
        </ProFormDependency>

        {/* 以下字段：仅在非 NGN/TZS/KES 时展示 */}
        <ProFormDependency name={['currencyType']}>
          {({ currencyType }) => {
            if (TIER2.includes(currencyType))
              return null

            return (
              <>
                <ProFormSelect
                  name="accAddrCountry"
                  label={t('accountDetail.form.countryRegion')}
                  placeholder={t('accountDetail.form.countryRegionPlaceholder')}
                  options={countryOptions}
                  rules={[{ required: true }]}
                />

                <ProFormText
                  name="bankAddress"
                  label={t('accountDetail.form.bankAddress')}
                  placeholder={t('accountDetail.form.bankAddressPlaceholder')}
                  rules={[{ required: true }]}
                />

                <ProFormText
                  name="bankSwift"
                  label={t('accountDetail.form.swift')}
                  placeholder={t('accountDetail.form.swiftPlaceholder')}
                  rules={[{ required: true }]}
                />

                <ProFormText
                  name="iban"
                  label={t('accountDetail.form.iban')}
                  placeholder={t('accountDetail.form.ibanPlaceholder')}
                />

                <ProFormText
                  name="bsbNumber"
                  label={t('accountDetail.form.bsbNumber')}
                  placeholder={t('accountDetail.form.bsbNumberPlaceholder')}
                />

                <ProFormText
                  name="routingNumber"
                  label={t('accountDetail.form.routingNumber')}
                  placeholder={t('accountDetail.form.routingNumberPlaceholder')}
                />

                <ProForm.Item
                  name="recipientAddress"
                  label={t('accountDetail.form.recipientAddress')}
                  rules={[{ required: true }]}
                >
                  <AddressWidget countries={countryList?.data ?? []} />
                </ProForm.Item>

                <ProFormText
                  name="payeePostCode"
                  label={t('accountDetail.form.postCode')}
                  placeholder={t('accountDetail.form.postCodePlaceholder')}
                  rules={[{ required: true }]}
                />
              </>
            )
          }}
        </ProFormDependency>

        {/* 上传附件 */}
        <ProForm.Item
          name="serviceAgreementFile"
          label={t('accountDetail.form.cooperationAgreement')}
          rules={[{ required: true }]}
        >
          <Upload />
        </ProForm.Item>

        <ProFormDependency name={['currencyType']}>
          {({ currencyType }) => (
            <ProForm.Item
              name="bankFlowFile"
              label={t('accountDetail.form.accountOpeningCertificate')}
              rules={[{ required: currencyType !== CURRENCY_TYPE_ENUM.NGN }]}
              extra={t('accountDetail.form.accountOpeningCertificateExtra')}
            >
              <Upload />
            </ProForm.Item>
          )}
        </ProFormDependency>

        <Button type="primary" htmlType="submit">
          {t('global.submit')}
        </Button>
      </ProForm>
    )
  }
```
